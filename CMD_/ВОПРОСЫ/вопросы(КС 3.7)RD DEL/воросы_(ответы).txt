Удаления каталогов:
1. rd    main
   rmdir main

2. rd    main1 main2
   rmdir main1 main2

3. rd /s main
   rd main /s
   
   rmdir /s main
   rmdir main /s

3.1 rd folder1 folder2 /s
    rmdir folder1 folder2 /s
    rd /s folder1 folder2
    rmdir /s folder1 folder2

4. rd /s /q main
   rd main /s /q

   rmdir /s /q main
   rmdir main /s /q

4.1 rd folder1 folder2 /s /q
    rmdir folder1 folder2 /s /q

    rd /s /q folder1 folder2
    rmdir /s /q folder1 folder2
